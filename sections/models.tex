\section[Representação Digital]{Representação Digital e Otimização}

\begin{frame}{Modelos Orientados ao Usuário: HSV e HSL}
    \begin{columns}
        \begin{column}{0.55\textwidth}
            \textbf{Problema do RGB (para Humanos):}
            \begin{itemize}
                \item O ser humano não calcula cores somando intensidades de luz.
            \end{itemize}

            \textbf{Abstração Matemática de Software:}
            \begin{itemize}
                \item Hue (Matiz).
                \item Saturation.
                \item Value / Lightness.
            \end{itemize}

            \textbf{Aplicação Prática (UI/UX):}
            \begin{itemize}
                \item Padrão para seletores de cor. O usuário escolhe a matiz e depois refina o brilho e a pureza.
            \end{itemize}
        \end{column}
        
        \begin{column}{0.45\textwidth}
            \begin{figure}
                \centering
                \includegraphics[width=0.9\textwidth]{images/cone_hsv.png}
                \caption{Espaço geométrico tridimensional do modelo HSV.}
            \end{figure}
        \end{column}
    \end{columns}

    \note{
        \begin{columns}[T]
            \begin{column}{0.49\textwidth}
                \scriptsize
                \textbf{RGB vs. HSV:} O RGB é um modelo voltado puramente ao hardware. O HSV/HSL são abstrações de software feitas para alinhar a computação à cognição humana.
                
                \textbf{Diferenças Técnicas Críticas:}
                \begin{itemize}
                    \setlength{\itemsep}{1pt}
                    \item \textbf{HSV:} $V=100\%$ representa a cor pura (muito usado no ecossistema Adobe/Figma).
                    \item \textbf{HSL:} A cor pura fica exatamente no meio do eixo, em $L=50\%$, pois em $L=100\%$ o modelo colapsa para branco absoluto. É o padrão nativo da Web (CSS).
                \end{itemize}
            \end{column}

            \begin{column}{0.51\textwidth}    
                \scriptsize
                \textbf{Engenharia de Software (UI):}
                \begin{itemize}
                    \setlength{\itemsep}{1pt}
                    \item \textbf{Conversão em Back-end:} O usuário manipula a cor de forma intuitiva em HSV no seletor, mas o programa converte os vetores para RGB em tempo real antes de despachar o sinal para o hardware.
                    \item \textbf{Aplicações em Código:} O HSL/HSV é fantástico na programação para gerar paletas consistentes e lógica de \textit{Dark Mode}, pois você pode alterar apenas a variável numérica de luminosidade ($L$) em loops aritméticos simples sem quebrar ou mudar a matiz original da identidade visual.
                \end{itemize}
            \end{column}
        \end{columns}   
    }
\end{frame}
