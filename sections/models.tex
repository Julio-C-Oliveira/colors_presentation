\section[Modelos]{Modelos de Cor e Transmissão}

\begin{frame}{Modelos Orientados ao Usuário: HSV e HSL}
    \begin{itemize}
        \item \textbf{Limitação do RGB:} O modelo RGB não é intuitivo para seres humanos (ex: é difícil adivinhar a composição RGB exata para um tom específico de marrom).
        \vspace{0.2cm}
        \item \textbf{Abstração Humana (HSV / HSL):}
        \begin{itemize}
            \item \textbf{H (Hue / Matiz):} Ângulo radial em um disco cromático ($0^\circ$ a $360^\circ$).
            \item \textbf{S (Saturation / Saturação):} Distância do eixo central (pureza).
            \item \textbf{V / L (Value ou Lightness / Brilho):} Eixo vertical de intensidade luminosa.
        \end{itemize}
        \vspace{0.2cm}
        \item \textbf{Representação Geométrica:} Mapeados como estruturas tridimensionais cônicas, cilíndricas ou hexagonais. Ideal para a implementação de seletores de cor (*Color Pickers*) em UIs.
    \end{itemize}

    \note{Enfatizar a diferença de arquitetura: RGB é orientado ao hardware (barramento, subpixels), enquanto HSV/HSL é orientado à cognição humana e artistas digitais. O programador faz a conversão matemática nos seletores em tempo real.}
\end{frame}

\begin{frame}{Transmissão e Processamento Cognitivo: YIQ e Cores Oponentes}
    \begin{itemize}
        \item \textbf{O Modelo YIQ (Padrão NTSC):}
        \begin{itemize}
            \item Criado para garantir a retrocompatibilidade da TV colorida com as TVs Preto e Branco.
            \item \textbf{Y (Luminância):} Intensidade da luz. O único canal lido por aparelhos P\&B.
            \item \textbf{I e Q (Crominância):} Informações de fase e amplitude da cor.
        \end{itemize}
        \vspace{0.2cm}
        \item \textbf{Teoria das Cores Oponentes:}
        \begin{itemize}
            \item O processamento neurológico no cérebro ocorre via canais antagonistas: \textbf{Vermelho-Verde}, \textbf{Azul-Amarelo} e \textbf{Preto-Branco}.
            \item Explica por que é neurologicamente impossível visualizar uma cor descrita como "verde-avermelhada".
            \item Origina o fenômeno de \textit{pós-imagem (afterimage)} após a fadiga de fotorreceptores específicos.
        \end{itemize}
    \end{itemize}

\end{frame}
