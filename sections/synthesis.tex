\section[Cores e Dispositivos]{Síntese de Cores e Dispositivos}

\begin{frame}{Síntese Aditiva: Modelo RGB}
    \begin{columns}
        \begin{column}{0.55\textwidth}
            \begin{itemize}
                \item \textbf{Sistemas Aditivos de Emissão:}
                \begin{itemize}
                    \item Aplicado em hardware emissor de luz diretamente para o olho: telas OLED, LCDs, projetores e TVs.
                \end{itemize}
                \vspace{0.2cm}
                \item \textbf{Subpixel:}
                \begin{itemize}
                    \item Cada pixel é uma junção de emissores independentes (\textbf{R}, \textbf{G}, \textbf{B}).
                \end{itemize}
                \vspace{0.2cm}
            \end{itemize}
        \end{column}
        
        \begin{column}{0.45\textwidth}
            \begin{figure}
                \centering
                \includegraphics[width=0.9\textwidth]{images/sintese_aditiva_rgb.jpg}
                \caption{Sobreposição de luzes RGB.}
            \end{figure}
        \end{column}
    \end{columns}

    \note{
        \begin{columns}[T]
            \begin{column}{0.49\textwidth}
                \scriptsize
                \textbf{Foco Prático e Hardware:}
                \begin{itemize}
                    \setlength{\itemsep}{1pt}
                    \item \textbf{Aparato Físico:} Monitores projetam fótons direto no olho, diferente de uma folha que absorve e reflete luz.
                    \item \textbf{Memória:} Na programação, um pixel de 24 bits dedica 1 byte (0 a 255) para controlar a voltagem/brilho de cada lâmpada.
                    \item \textbf{Controle Elétrico:} Cortar a energia do barramento gera o preto absoluto (muito nítido em telas OLED onde o LED apaga individualmente).
                \end{itemize}
            \end{column}

            \begin{column}{0.49\textwidth}    
                \scriptsize
                \textbf{O Barramento ($C = rR + gG + bB$):}
                \begin{itemize}
                    \setlength{\itemsep}{1pt}
                    \item \textbf{O que é:} Tradução exata de como o Software (seu código) controla o Hardware (os LEDs da tela).
                    \item \textbf{Fixos ($R, G, B$):} Três subpixels físicos (micro-LEDs) soldados na placa do monitor.
                    \item \textbf{Variáveis ($r, g, b$):} Valores digitais definidos na memória (ex: 0 a 255 ou 0.0 a 1.0).
                    \item \textbf{Mecanismo:} Funcionam como reguladores de voltagem. O barramento envia os números e a tela injeta corrente.
                    \item \textbf{Exemplo:} Se o código define $r=255, g=0, b=0$, manda energia máxima para o vermelho e desliga os outros.
                \end{itemize}
            \end{column}
        \end{columns}
    }
\end{frame}

    \note{
        \begin{columns}[T]
            \begin{column}{0.49\textwidth}
                \scriptsize
                \textbf{Análise da Imagem Ampliada:}
                \begin{itemize}
                    \setlength{\itemsep}{1pt}
                    \item \textbf{A Ilusão Óptica:} O que o usuário enxerga a distância como uma imagem contínua e suave, na verdade é uma matriz discreta de emissores isolados.
                    \item \textbf{Geometria do Pixel:} Cada ponto luminoso (pixel) é composto pela tríade física fixa: uma faixa Vermelha, uma Verde e uma Azul.
                    \item \textbf{Cores Secundárias:} Para gerar Amarelo, o hardware acende os filetes R e G com potência máxima e desliga o B. A mistura ocorre no espaço (retina), não na tela.
                \end{itemize}
            \end{column}

            \begin{column}{0.49\textwidth}    
                \scriptsize
                \textbf{Conexão com Baixo Nível:}
                \begin{itemize}
                    \setlength{\itemsep}{1pt}
                    \item \textbf{Controle de Intensidade:} A variação de cores que vemos na imagem não é o LED mudando de cor, mas sim a variação do brilho relativo (luminância) de cada filete.
                    \item \textbf{Densidade (PPI):} Telas modernas possuem centenas de pixels por polegada, tornando esses filetes imperceptíveis a olho nu, exigindo lentes ou macro.
                    \item \textbf{Analogia Técnica:} O Framebuffer na memória dita diretamente a modulação por largura de pulso (PWM) ou a tensão elétrica que alimenta cada um desses filetes individuais.
                \end{itemize}
            \end{column}
        \end{columns}
    }
\end{frame}

\begin{frame}{Síntese Subtrativa: Modelo CMYK}
    \begin{columns}
        \begin{column}{0.55\textwidth}
            \textbf{Sistemas Subtrativos de Absorção:}
            \begin{itemize}
                \item Aplicado em meios físicos onde a cor \textbf{não tem luz própria} (papel, tintas, impressos).
                \item A tinta funciona como um filtro físico: ela \textbf{subtrai (absorve)} frequências da luz ambiente e reflete apenas o resto para o nosso olho.
            \end{itemize}
        \end{column}
        
        \begin{column}{0.45\textwidth}
            \begin{figure}
                \centering
                \includegraphics[width=0.9\textwidth]{images/sintese_subtrativa_cmyk.png}
                \caption{Subtração de luz - CMYK.}
            \end{figure}
        \end{column}
    \end{columns}

    \note{
        \begin{columns}[T]
            \begin{column}{0.5\textwidth}
                \scriptsize
                \textbf{Física dos Pigmentos Primários:}
                \begin{itemize}
                    \item \textbf{Ciano} absorve Vermelho (reflete G+B).
                    \item \textbf{Magenta} absorve Verde (reflete R+B).
                    \item \textbf{Amarelo} absorve Azul (reflete R+G).
                \end{itemize}
                \vspace{0.1cm}
                \textbf{A Lógica Prática:}
                \begin{itemize}
                    \setlength{\itemsep}{1pt}
                    \item \textbf{Filtros Químicos:} Quando você joga tinta ciano no papel, está criando uma barreira química que "engole" os fótons vermelhos da lâmpada do quarto. O que sobra e volta pro seu olho é o Verde e Azul, que juntos formam o Ciano.
                \end{itemize}
            \end{column}

            \begin{column}{0.5\textwidth}    
                \scriptsize
                \textbf{Engenharia Gráfica e o Canal K:}
                \begin{itemize}
                    \setlength{\itemsep}{1pt}
                    \item \textbf{O mistério do K:} Significa \textit{Key Plate} (a placa matriz preta que dava os detalhes de contorno na prensa tradicional).
                    \item \textbf{Problema de Engenharia Química:} Misturar $C+M+Y$ puramente cria um marrom lamacento devido a impurezas dos pigmentos reais.
                    \item \textbf{Custo e Secagem:} Imprimir textos longos usando CMY para simular o preto encharcaria o papel (ele enruga ou rasga) e gastaria 3x mais cartucho. A tinta preta pura ($K$) resolve esse problema operacional.
                \end{itemize}
            \end{column}
        \end{columns}
    }
\end{frame}

\begin{frame}{O Conflito de Gamut}
    \begin{columns}
        \begin{column}{0.55\textwidth}
            \begin{itemize}
                \item \textbf{O Problema do Mundo Real:}
                \begin{itemize}
                    \item \textbf{Gamut} é o limite físico de cores que um determinado hardware consegue reproduzir.
                \end{itemize}
                \vspace{0.2cm}
                \item \textbf{A Solução via Software:}
                \begin{itemize}
                    \item Quando uma cor digital da tela cai "fora" do limite físico da impressora, o sistema operacional roda algoritmos de \textbf{Mapeamento de Gamut} para traduzir a cor da forma mais fiel possível.
                \end{itemize}
            \end{itemize}
        \end{column}
        
        \begin{column}{0.45\textwidth}
            \begin{figure}
                \centering
                \includegraphics[width=0.9\textwidth]{images/grafico_gamut_cie.png}
                \caption{Interseção e distorção entre os limites RGB e CMYK.}
            \end{figure}
        \end{column}
    \end{columns}

    \note{
        \begin{columns}[T]
            \begin{column}{0.52\textwidth}
                \scriptsize
                \textbf{Leitura do Gráfico (Diagrama CIE):}
                \begin{itemize}
                    \setlength{\itemsep}{0pt}
                    \item \textbf{A Ferradura:} Todo o espectro visível pelo olho humano (visto na Seção 1).
                    \item \textbf{Triângulo (RGB):} O limite que as telas alcançam.
                    \item \textbf{Polígono interno (CMYK):} O limite da impressão.
                    \item \textbf{O Choque:} A área do triângulo RGB que fica de fora do CMYK são as cores impossíveis de imprimir (o famoso susto do cliente: verde neon na tela vira verde fosco no papel).
                \end{itemize}
            \end{column}

            \begin{column}{0.48\textwidth}    
                \scriptsize
                \textbf{Estratégias de Renderização (CMS):}
                \begin{itemize}
                    \setlength{\itemsep}{1pt}
                    \item \textbf{Perceptual (Fotográfico):} Reduz o gamut inteiro proporcionalmente. Comprime todas as cores da imagem (inclusive as válidas) para preservar a relação visual entre elas e evitar "estouros" de degradê.
                    \item \textbf{Colorimétrico Relativo:} Faz um mapeamento por poda (\textit{clipping}). Mantém as cores internas idênticas. Tudo o que estiver fora do limite é achatado (\textit{clamped}) na borda mais próxima. Ótimo para logotipos, mas destrói degradês finos.
                \end{itemize}
            \end{column}
        \end{columns}   
    }
\end{frame}
